スキルチェック  
Bランク　1問  
  
-------------------------------------------
  
・4桁のプレート番号が3つ与えられる  
・各プレートの各桁に+か-を入れて算出した数値を記録    
・全プレートに共通値があればYESと、共通値を小〜大に出力  
・なければNOと出力  

-------------------------------------------
  
```ruby
plates = 3.times.flat_map{gets.split.map(&:chars)}   # 3つのプレート番号を取得

result = []                                          # 各プレートの結果を記録用
plates.each do |plate|                               # 各プレートを計算していく
  plate = plate.map(&:to_i)                          # 各桁を数値化
  all = []                                           # 全可能性の結果を記録
  all << plate[0]+plate[1]+plate[2]+plate[3]         # 全8種類の可能性を算出
  all << plate[0]+plate[1]+plate[2]-plate[3]
  all << plate[0]+plate[1]-plate[2]+plate[3]
  all << plate[0]-plate[1]+plate[2]+plate[3]
  all << plate[0]+plate[1]-plate[2]-plate[3]
  all << plate[0]-plate[1]+plate[2]-plate[3]
  all << plate[0]-plate[1]-plate[2]+plate[3]
  all << plate[0]-plate[1]-plate[2]-plate[3]
  result << all                                     # プレートごとの結果を記録
end

same = result[0]&result[1]&result[2]                # 全プレート共通値を定義

if same.size != 0                                   # 共通値あればYESと、共通値を小〜大に出力
  puts "YES"
  puts same.sort.join(" ")
else
  puts "NO"                                         # なければNOと出力
end
```


もっと簡潔にできる
```ruby
plates = 3.times.map { gets.chars.map(&:to_i) }       # 3つのプレート番号を取得

patterns = ['+', '-'].repeated_permutation(3).to_a    # 演算子の全可能性をarray化(+ と - が計3箇所、全8パターン)

results = []

plates.each do |digits|)                              # 各プレートを計算していく
  values = []                                         # 各プレートの結果を記録用
  patterns.each do |ops|)                             # 演算子の各パターンを代入
    expr = "#{digits[0]}#{ops[0]}
            #{digits[1]}#{ops[1]}
            #{digits[2]}#{ops[2]}
            #{digits[3]}"
    values << eval(expr)                              # 文字列をコードとして計算して記録
  end

  results << values                                   # 算出した結果を記録
end

common = results.reduce(:&)                           # 各プレートの共通値を定義

if common.empty?                                      # 共通値なければNOと出力
  puts "NO"
else                                                  # あればYESと、共通値を小〜大に出力
  puts "YES"
  puts common.sort.join(" ")
end
```
